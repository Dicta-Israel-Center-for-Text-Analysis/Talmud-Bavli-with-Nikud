 *** דף ג. *** 

זֹאת תּוֹרַת הָעוֹלָה הִיא הֲרֵי אֵלּוּ שְׁלֹשָׁה מִעוּטִין
וְאִיבָּעֵית אֵימָא עֲדַיִן אֲנִי אוֹמֵר לָא מָצֵית מוֹקְמַתְּ לַהּ כְּרַבִּי יְהוּדָה דְּקָתָנֵי רֹב קָהָל שֶׁחָטְאוּ בֵּית דִּין מְבִיאִין עַל יְדֵיהֶן פַּר וְאִי רַבִּי יְהוּדָה הָאָמַר צִבּוּר הוּא דְּמַיְתֵי בֵּית דִּין לָא דִּתְנַן רַבִּי יְהוּדָה אוֹמֵר שִׁבְעָה שְׁבָטִים שֶׁחָטְאוּ מְבִיאִין שִׁבְעָה פָּרִים

בִּשְׁלָמָא לְרַבִּי יְהוּדָה מַשְׁכַּחַתְּ לַהּ לְהָנֵי שְׁנֵים עָשָׂר חַטָּאוֹת כְּגוֹן דְּחָטְאוּ שְׁנֵים עָשָׂר שְׁבָטִים דְּמַיְתוּ שְׁנֵים עָשָׂר שְׂעִירִים אִי נָמֵי דְּחָטְאוּ שִׁבְעָה שְׁבָטִים וּשְׁאָרָא אִינָךְ בִּגְרִירָה וּלְרַבִּי שִׁמְעוֹן נָמֵי מַשְׁכַּחַתְּ לַהּ כְּגוֹן דְּחָטְאוּ אַחַד עָשָׂר שְׁבָטִים דְּמַיְתוּ אַחַד עָשָׂר שְׂעִירִים וְאִדַּךְ דְּבֵית דִּין אֶלָּא לְרַבִּי מֵאִיר דְּאָמַר בֵּית דִּין מְבִיאִין וְלֹא צִבּוּר שְׁנֵים עָשָׂר הֵיכִי מַשְׁכַּחַתְּ לֵהּ כְּגוֹן דְּחָטְאוּ וַהֲדַר חָטְאוּ וַהֲדַר חָטְאוּ עַד תְּרֵיסַר זִמְנֵי
וְהָא מַיְתֵי לְהוּ הָנְהוּ דְּחָטְאוּ
אָמַר רַב פָּפָּא כִּי גְּמִירִי חַטָּאת שֶׁמֵּתוּ בְּעָלֶיהָ בְּמִיתָה הָנֵי מִלֵּי בְּיָחִיד אֲבָל לֹא בְּצִבּוּר לְפִי שֶׁאֵין מִיתָה בְּצִבּוּר מְנָא לֵהּ לְרַב פָּפָּא הָא אִלֵּימָא מִדִּכְתִיב תַּחַת אֲבֹתֶיךָ יִהְיוּ בָנֶיךָ אִי הָכִי אֲפִלּוּ בְּיָחִיד נָמֵי
אֶלָּא דֻּקְיָא דְּרַב פָּפָּא מִשָּׂעִיר דְּרֹאשׁ חֹדֶשׁ דְּאָמַר רַחֲמָנָא מַיְתֵי מִתְּרוּמַת הַלִּשְׁכָּה וְהָא מַיְתֵי לְהוּ מִיִּשְׂרָאֵל וְהָנָךְ דְּפָיְשִׁי הֵיכִי מַיְתוּ אֶלָּא שְׁמַע מִנַּהּ חַטָּאת שֶׁמֵּתוּ בְּעָלֶיהָ בְּצִבּוּר קְרֵבָה
מִי דָּמֵי שָׂעִיר דְּרֹאשׁ חֹדֶשׁ דִּלְמָא לָא מַיְתוּ מִצִּבּוּר אֲבָל הָכָא וַדַּאי מַיְתוּ אֶלָּא טַעְמָא דְּרַב פָּפָּא מֵהָכָא דִּכְתִיב כַּפֵּר לְעַמְּךָ יִשְׂרָאֵל אֲשֶׁר פָּדִיתָ ה' רְאוּיָה כַּפָּרָה זוֹ שֶׁתְּכַפֵּר עַל יוֹצְאֵי מִצְרַיִם מִדִּכְתִיב אֲשֶׁר פָּדִיתָ
מִי דָּמֵי הָתָם כֻּלְּהוּ אִיתִינוּן מִגּוֹ דִּמְכַפְּרָה אַחַיִּים מְכַפְּרָה נָמֵי אַמֵּתִים אֶלָּא הָכָא מִי הֲווֹ חַיִּים אִין הָכִי נָמֵי דִּכְתִיב וְרַבִּים מֵהַכֹּהֲנִים וְהַלְוִיִּם וְרָאשֵׁי הָאָבוֹת וְגוֹ'
וְדִלְמָא מוּעָטִין הֲווֹ וְלָא רַבִּים הֲווֹ הָכְתִיב (וְלֹא הִכִּירוּ בְּקוֹל) [וְאֵין הָעָם מַכִּירִים קוֹל] תְּרוּעַת הַשִּׂמְחָה לְקוֹל בְּכִי הָעָם [וְגוֹ'] וְהַקּוֹל נִשְׁמַע עַד לְמֵרָחוֹק
וְהָא מְזִידִין הֲווֹ הוֹרָאַת שָׁעָה הָיְתָה הָכִי נָמֵי מִסְתַּבְּרָא דְּאִי לָא תֵּימָא הָכִי אֵילִים תִּשְׁעִים וְשִׁשָּׁה כְּבָשִׂים שִׁבְעִים וְשִׁבְעָה כְּנֶגֶד מִי אֶלָּא הוֹרָאַת שָׁעָה הָיְתָה הָכָא נָמֵי הוֹרָאַת שָׁעָה הָיְתָה
תָּנוּ רַבָּנַן מֵת אֶחָד מִן הַצִּבּוּר חַיָּבִין אֶחָד מִבֵּית דִּין פְּטוּרִין מַאן תַּנָּא אָמַר רַב חִסְדָּא אָמַר רַבִּי זֵירָא אָמַר רַב יִרְמְיָה אָמַר רַב רַבִּי מֵאִיר הִיא דְּאָמַר בֵּית דִּין מְבִיאִים וְלֹא צִבּוּר הִלְכָּךְ מֵת אֶחָד מִן הַצִּבּוּר חַיָּבִין דְּהָא קָאֵם כֻּלֵּהּ בֵּית דִּין מֵת אֶחָד מִבֵּית דִּין פְּטוּרִין דְּהָוְיָא לַהּ חַטָּאת שֶׁמֵּת אֶחָד מִן הַשֻּׁתָּפִין וּמִשּׁוּם הָכִי פְּטוּרִין
מַתְקֵף לַהּ רַב יוֹסֵף וְנוֹקְמַהּ כְּרַבִּי שִׁמְעוֹן דְּאָמַר בֵּית דִּין עִם הַצִּבּוּר מֵת אֶחָד מִן הַצִּבּוּר חַיָּבִין דְּאֵין צִבּוּר מֵתִים מֵת אֶחָד מִבֵּית דִּין פְּטוּרִין כִּדְאָמְרִינַן דְּחַטָּאת שֻׁתָּפִין הִיא
אֲמַר לֵהּ אַבָּיֵי שָׁמְעִינַן לֵהּ לְרַבִּי שִׁמְעוֹן דְּאָמַר חַטַּאת שֻׁתָּפִין אֵינָהּ מֵתָה דְּתַנְיָא פַּר וְשָׂעִיר שֶׁל יוֹם הַכִּפּוּרִים שֶׁאָבְדוּ וְהִפְרִישׁ אֲחֵרִים תַּחְתֵּיהֶם יָמוּתוּ כֻּלָּן דִּבְרֵי רַבִּי יְהוּדָה רַבִּי אֶלְעָזָר וְרַבִּי שִׁמְעוֹן אוֹמֵר יִרְעוּ לְפִי שֶׁאֵין חַטַּאת צִבּוּר מֵתָה
אֲמַר לֵהּ רַב יוֹסֵף כֹּהֲנִים קָא אָמְרַתְּ שָׁאנֵי כֹּהֲנִים דְּאִקְּרוּ קָהָל דִּכְתִיב עַל הַכֹּהֲנִים וְעַל כׇּל עַם הַקָּהָל יְכַפֵּר

 *** דף ו: *** 

אֶלָּא מֵעַתָּה נַיְתוֹ פַּר בְּהוֹרָאָה וְכִי תֵּימָא הָכִי נָמֵי טָפִי לְהוּ שְׁבָטִים
אֶלָּא אָמַר רַב אַחָא בְּרַבִּי יַעֲקֹב שִׁבְטוֹ שֶׁל לֵוִי לָא אִקְּרוּ קָהָל דִּכְתִיב הִנְנִי מַפְרְךָ וְהִרְבִּיתִךָ וּנְתַתִּיךָ לִקְהַל עַמִּים וְגוֹ' כֹּל שֶׁיֵּשׁ לוֹ אֲחֻזָּה אִקְּרִי קָהָל וְכֹל שֶׁאֵין לוֹ אֲחֻזָּה לָא אִקְּרִי קָהָל
אִם כֵּן חָסְרִי לְהוּ שְׁנֵים עָשָׂר שְׁבָטִים אָמַר אַבָּיֵי אֶפְרַיִם וּמְנַשֶּׁה כִּרְאוּבֵן וְשִׁמְעוֹן יִהְיוּ לִי אָמַר רָבָא וְהָא כְּתִיב עַל שֵׁם אֲחֵיהֶם יִקָּרְאוּ בְּנַחֲלָתָם לְנַחֲלָה הֻקְשׁוּ וְלֹא לְדָבָר אַחֵר
וְלָא וְהָא חֲלוּקִין בַּדְּגָלִים כְּנַחֲלָתָן כָּךְ חֲנִיָּתָן כְּדֵי לְחַלֵּק כָּבוֹד לַדְּגָלִים
וְהָא חֲלוּקִים בַּנְּשִׂיאִים הָהוּא לַחֲלֹק כָּבוֹד לַנְּשִׂיאִים דְּתַנְיָא שְׁלֹמֹה עָשָׂה שִׁבְעָה יְמֵי חֲנֻכָּה וּמָה רָאָה מֹשֶׁה לַעֲשׂוֹת שְׁנֵים עָשָׂר יְמֵי חֲנֻכָּה כְּדֵי לַחֲלֹק כָּבוֹד לַנְּשִׂיאִים
מַאי הָוֵי עֲלַהּ
תָּא שְׁמַע דְּתַנְיָא רַבִּי שִׁמְעוֹן אוֹמֵר חָמֵשׁ חַטָּאוֹת מֵתוֹת וְלַד חַטָּאת וּתְמוּרַת חַטָּאת וְחַטָּאת שֶׁמֵּתוּ בְּעָלֶיהָ וְחַטָּאת שֶׁנִּתְכַּפְּרוּ בְּעָלֶיהָ וְחַטָּאת שֶׁעָבְרָה שְׁנָתָהּ וְאִי אַתָּה יָכוֹל לוֹמַר וְלַד חַטָּאת בְּצִבּוּר שֶׁאֵין צִבּוּר מַפְרִישִׁין נְקֵבָה וְאִי אַתָּה יָכוֹל לוֹמַר תְּמוּרַת חַטָּאת בְּצִבּוּר שֶׁאֵין תְּמוּרָה בְּצִבּוּר וְאִי אַתָּה יָכוֹל לוֹמַר חַטָּאת שֶׁמֵּתוּ בְּעָלֶיהָ בְּצִבּוּר שֶׁאֵין צִבּוּר מֵתִים
שֶׁנִּתְכַּפְּרוּ בְּעָלֶיהָ וְשֶׁעָבְרָה שְׁנָתָהּ לֹא שָׁמַעְנוּ יָכוֹל יָמוּתוּ אָמַרְתָּ יִלְמֹד סָתוּם מִן הַמְפֹרָשׁ מַה מָּצִינוּ בִּוְלַד חַטָּאת וּתְמוּרַת חַטָּאת וְשֶׁמֵּתוּ בְּעָלֶיהָ בְּיָחִיד דְּבָרִים אֲמוּרִים וְלֹא בְּצִבּוּר אַף שֶׁנִּתְכַּפְּרוּ בְּעָלֶיהָ וְשֶׁעָבְרָה שְׁנָתָהּ בְּיָחִיד דְּבָרִים אֲמוּרִים וְלֹא בְּצִבּוּר

הַדְרָן עֲלָךְ הוֹרוּ בֵּית דִּין

לְמִן הַיּוֹם אֲשֶׁר צִוָּה ה' וָהָלְאָה לְדֹרֹתֵיכֶם אֵיזוֹ הִיא מִצְוָה שֶׁהִיא נֶאֶמְרָה בִּתְחִלָּה הֱוֵי אוֹמֵר זוֹ עֲבוֹדָה זָרָה וְהָא אָמַר מָר עֶשֶׂר מִצְוֹת נִצְטַוּוּ יִשְׂרָאֵל בְּמָרָה (דִּכְתִיב וְהָיָה אִם שָׁמוֹעַ תִּשְׁמַע לְקוֹל ה' אֱלֹהֶיךָ) אֶלָּא מְחַוַּרְתָּא כִדְשַׁנִּינַן מֵעִקָּרָא
מַתְנִי' אֵין חַיָּבִין עַל עֲשֵׂה וְעַל לֹא תַעֲשֶׂה שֶׁבַּמִּקְדָּשׁ וְאֵין מְבִיאִין אָשָׁם תָּלוּי עַל עֲשֵׂה וְעַל לֹא תַעֲשֶׂה שֶׁבַּמִּקְדָּשׁ
אֲבָל חַיָּבִין עַל עֲשֵׂה וְעַל לֹא תַעֲשֶׂה שֶׁבַּנִּדָּה וּמְבִיאִים אָשָׁם תָּלוּי עַל עֲשֵׂה וְעַל לֹא תַעֲשֶׂה שֶׁבַּנִּדָּה אֵיזוֹ הִיא מִצְוַת עֲשֵׂה שֶׁבַּנִּדָּה פְּרֹשׁ מִן הַנִּדָּה וּמִצְוַת לֹא תַעֲשֶׂה אַל תָּבֹא עַל הַנִּדָּה
גְּמָ' מְנָא הָנֵי מִלֵּי דְּצִבּוּר לָא מְחַיְּבִי קׇרְבָּן בְּעָלְמָא וְיָחִיד אָשָׁם תָּלוּי נָמֵי לָא
אָמַר רַב יִצְחָק בַּר אַבְדִּימִי נֶאֱמַר וְאָשֵׁם בְּחַטָּאת וּבְאָשָׁם תָּלוּי וְנֶאֱמַר וְאָשֵׁמוּ בְּצִבּוּר מָה וְאָשֵׁם בְּיָחִיד בְּחַטָּאת קְבוּעָה אַף וְאָשֵׁמוּ בְּצִבּוּר בְּחַטָּאת קְבוּעָה
וּמָה צִבּוּר בְּחַטָּאת קְבוּעָה אַף אָשָׁם תָּלוּי נָמֵי אֵין בָּא אֶלָּא עַל סְפֵק חַטָּאת קְבוּעָה אָמְרִי אִי הָכִי קׇרְבָּן עוֹלֶה וְיוֹרֵד נָמֵי הָא כְּתִיב וְהָיָה כִּי יֶאְשַׁם לְאַחַת מֵאֵלֶּה דָּנִין (אָשֵׁם מִן אָשֵׁמוּ) [אָשֵׁמוּ מִן אָשֵׁם] וְאֵין דָּנִין אָשֵׁם מִן יֶאְשַׁם
וּמַאי נָפְקָא מִנַּהּ וְהָא תָּנֵי דְּבֵי רַבִּי יִשְׁמָעֵאל וְשָׁב הַכֹּהֵן וּבָא הַכֹּהֵן זוֹ הִיא שִׁיבָה זוֹ הִיא בִּיאָה וְעוֹד נֵילַף מִן וְאָשֵׁם מִטֻּמְאַת מִקְדָּשׁ וְקָדָשָׁיו דִּכְתִיב וְהוּא טָמֵא וְאָשֵׁם
אָמַר רַב פָּפָּא דָּנִין וְאָשֵׁם וּמִצְוֹת ה' מִן וְאָשֵׁם וּמִצְוֹת ה' אֲמַר לֵהּ רַב שִׁימִי בַּר אָשֵׁי לְרַב פָּפָּא וְנֵילַף וְאָשֵׁם וּנְשִׂיאַת עָוֹן מִן וְאָשֵׁם וּנְשִׂיאַת עָוֹן
אֶלָּא אָמַר רַב נַחְמָן בַּר יִצְחָק דָּנִין וְאָשֵׁם וּמִצְוֹת ה' אֲשֶׁר לֹא תֵעָשֶׂינָה מוְאָשֵׁם וּמִצְוֹת ה' אֲשֶׁר לֹא תֵעָשֶׂינָה וְאַל יוֹכִיחַ שְׁמִיעַת קוֹל וּבִטּוּי שְׂפָתַיִם וְטֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו שֶׁלֹּא נֶאֱמַר בָּהֶם וְאָשֵׁם וּמִצְוֹת ה' אֲשֶׁר לֹא תֵעָשֶׂינָה
מַתְנִי' אֵין חַיָּבִין עַל שְׁמִיעַת קוֹל וְעַל בִּטּוּי שְׂפָתַיִם וְעַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו וְהַנָּשִׂיא כַּיּוֹצֵא בָּהֶם דִּבְרֵי רַבִּי יוֹסֵי הַגְּלִילִי רַבִּי עֲקִיבָא אוֹמֵר הַנָּשִׂיא חַיָּב בְּכֻלָּן חוּץ מִשְּׁמִיעַת הַקּוֹל שֶׁהַמֶּלֶךְ לֹא דָּן וְלֹא דָּנִין אוֹתוֹ [לֹא מֵעִיד וְלֹא מְעִידִין אוֹתוֹ]
גְּמָ' אָמַר עֻלָּא מַאי טַעְמָא דְּרַבִּי יוֹסֵי הַגְּלִילִי אָמַר קְרָא וְהָיָה כִי יֶאְשַׁם לְאַחַת מֵאֵלֶּה כֹּל שֶׁמִּתְחַיֵּב בְּאַחַת מִתְחַיֵּב בְּכֻלָּן וְשֶׁאֵין מִתְחַיֵּב בְּאַחַת אֵין מִתְחַיֵּב בְּכֻלָּן
וְאֵימָא מִתְחַיֵּב בְּאַחַת מֵהֶן וְאַף עַל פִּי שֶׁאֵין מִתְחַיֵּב בְּכֻלָּן
אֶלָּא טַעְמָא דְּרַבִּי יוֹסֵי הַגְּלִילִי מֵהָכָא דְּתַנְיָא הָיָה רַבִּי יִרְמְיָה אוֹמֵר נֶאֱמַר

 *** דף ט. *** 

לֹא תַגִּיעַ יָדוֹ וְנֶאֱמַר לֹא תַשִּׂיג יָדוֹ מִי שֶׁבָּא לִידֵי עֲנִיּוּת וַעֲשִׁירוּת יָצָא נָשִׂיא וּמָשִׁיחַ שֶׁאֵין בָּאִין לִידֵי עֲנִיּוּת
נָשִׂיא דִּכְתִיב וְעָשָׂה אַחַת מִכׇּל מִצְוֹת ה' אֱלֹהָיו מִי שֶׁאֵין עַל גַּבָּיו אֶלָּא ה' אֱלֹהָיו מָשִׁיחַ דִּכְתִיב וְהַכֹּהֵן הַגָּדוֹל מֵאֶחָיו שֶׁהוּא גָּדוֹל מֵאֶחָיו בְּנוֹי בְּכֹחַ בְּחָכְמָה וּבְעֹשֶׁר אֲחֵרִים אוֹמְרִים מִנַּיִן שֶׁאִם אֵין לוֹ גַּדְּלֵהוּ מִשֶּׁל אֶחָיו תַּלְמוּד לוֹמַר וְהַכֹּהֵן הַגָּדוֹל מֵאֶחָיו אֲשֶׁר יוּצַק עַל רֹאשׁוֹ גַּדְּלֵהוּ מֵאֶחָיו
בְּעָא מִנֵּהּ רָבִינָא מֵרַב נַחְמָן בַּר יִצְחָק נָשִׂיא שֶׁנִּצְטָרַע מַהוּ מִדְחָא דְּחֵי אוֹ מִפְטָר פְּטִיר אֲמַר לֵהּ דִּילָךְ אוֹ דְּגַזָּא
תַּנְיָא רַבִּי עֲקִיבָא אוֹמֵר מָשִׁיחַ פָּטוּר מִכֻּלָּן אָמַר רָבָא מַאי טַעְמָא דְּרַבִּי עֲקִיבָא אָמַר קְרָא זֶה קׇרְבַּן אַהֲרֹן וּבָנָיו זוֹ בָּאָה חוֹבָה לוֹ וְאֵין אַחֶרֶת בָּאָה חוֹבָה לוֹ
וְאֵימָא כִּי מְמַעֵט לֵהּ רַחֲמָנָא מִדַּלֵּי דַלּוּת וּמַאי נִיהוּ עֲשִׂרִית הָאֵיפָה אֲבָל עֲנִיּוּת וַעֲשִׁירוּת לָא מַעֲטֵהּ רַחֲמָנָא לָא סָלְקָא דַּעְתָּךְ דִּכְתִיב וְכִפֶּר עָלָיו הַכֹּהֵן עַל חַטָּאתוֹ אֲשֶׁר חָטָא מֵאַחַת מֵאֵלֶּה הַמִּתְכַּפֵּר בְּאַחַת מִתְכַּפֵּר בְּכֻלָּן וְשֶׁאֵין מִתְכַּפֵּר בְּאַחַת אֵין מִתְכַּפֵּר בְּכֻלָּן
אֶלָּא מֵעַתָּה דִּכְתִיב וְהָיָה כִּי יֶאְשַׁם לְאַחַת מֵאֵלֶּה הָכִי נָמֵי דְּכׇל הַמִּתְחַיֵּב בְּאַחַת מִתְחַיֵּב בְּכֻלָּן וְשֶׁאֵין מִתְחַיֵּב בְּאַחַת אֵין מִתְחַיֵּב בְּכֻלָּן אַלְּמָה תְּנַן רַבִּי עֲקִיבָא אוֹמֵר נָשִׂיא חַיָּב חוּץ מִשְּׁמִיעַת קוֹל
אַבָּיֵי וְרָבָא דְּאָמְרִי תַּרְוַיְהוּ מֵאַחַת מַשְׁמַע לֵהּ לְאַחַת לָא מַשְׁמַע לֵהּ וּמַאי שְׁנָא מֵאַחַת דְּמַשְׁמַע לֵהּ
דְּכַתְבֵהּ רַחֲמָנָא לְבַסּוֹף גַּבֵּי עֲשִׂרִית הָאֵיפָה לְמֵימְרָא דְּכֹל דְּמִחַיַּב בַּעֲשִׂרִית הָאֵיפָה מִחַיַּב בְּכֻלָּן דְּאִי סָלְקָא דַעְתָּךְ מִתְחַיֵּב בְּאַחַת אַף עַל פִּי שֶׁאֵין מִתְחַיֵּב בְּכֻלָּן נִכְתְּבֵהּ לְהַאי מֵאַחַת מֵאֵלֶּה בְּדַלּוּת אִי נָמֵי בַּעֲשִׁירוּת
מַתְנִי' כׇּל הַמִּצְוֹת שֶׁבְּתוֹרָה שֶׁחַיָּבִין עַל זְדוֹנָן כָּרֵת וְעַל שִׁגְגָתָן חַטָּאת הַיָּחִיד מֵבִיא כִּשְׂבָּה וּשְׂעִירָה וְהַנָּשִׂיא שָׂעִיר וּמָשִׁיחַ וּבֵית דִּין מְבִיאִין פַּר וּבַעֲבוֹדָה זָרָה הַיָּחִיד וְהַנָּשִׂיא וְהַמָּשִׁיחַ מְבִיאִין שְׂעִירָה וּבֵית דִּין פַּר וְשָׂעִיר פַּר לְעוֹלָה וְשָׂעִיר לְחַטָּאת
אָשָׁם תָּלוּי הַיָּחִיד וְהַנָּשִׂיא חַיָּבִין וּמָשִׁיחַ וּבֵית דִּין פְּטוּרִין אָשָׁם וַדַּאי הַיָּחִיד וְהַנָּשִׂיא וְהַמָּשִׁיחַ חַיָּבִין וּבֵית דִּין פְּטוּרִין

 *** דף יד. *** 

לֹא הָיָה קָרֵב אָמַר לוֹ מִי הֵם הַלָּלוּ שֶׁמֵּימֵיהֶם אָנוּ שׁוֹתִים וּשְׁמוֹתָם אֵין אָנוּ מַזְכִּירִים אֲמַר לֵהּ בְּנֵי אָדָם שֶׁבִּקְּשׁוּ לַעֲקֹר כְּבוֹדְךָ וּכְבוֹד בֵּית אָבִיךָ
אֲמַר לֵהּ גַּם אַהֲבָתָם גַּם שִׂנְאָתָם גַּם קִנְאָתָם כְּבָר אָבָדָה אֲמַר לֵהּ הָאוֹיֵב תַּמּוּ חֳרָבוֹת לָנֶצַח אֲמַר לֵהּ הָנֵי מִלֵּי הֵיכָא דְּאַהֲנוֹ מַעֲשַׂיְהוּ רַבָּנַן לָא אַהֲנוֹ מַעֲשַׂיְהוּ הֲדַר אַתְנִי לֵהּ אָמְרוּ מִשּׁוּם רַבִּי מֵאִיר אִלּוּ הָיָה תְּמוּרָה לֹא הָיָה קָרֵב אָמַר רָבָא אֲפִלּוּ רַבִּי דְּעִנְוְתָנָא הוּא (תְּנָא) אָמְרוּ מִשּׁוּם רַבִּי מֵאִיר אָמַר רַבִּי מֵאִיר לָא אָמַר
